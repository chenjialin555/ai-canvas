放：

```ts
export type Point = {
  x: number;
  y: number;
};

export type Viewport = {
  zoom: number;
  pan: Point;
};

export function screenToWorld(point: Point, viewport: Viewport): Point {
  return {
    x: (point.x - viewport.pan.x) / viewport.zoom,
    y: (point.y - viewport.pan.y) / viewport.zoom,
  };
}

export function worldToScreen(point: Point, viewport: Viewport): Point {
  return {
    x: point.x * viewport.zoom + viewport.pan.x,
    y: point.y * viewport.zoom + viewport.pan.y,
  };
}
```

然后把 `StageCanvas.tsx` 里类似：

const x = (point.x - pan.x) / zoom;
const y = (point.y - pan.y) / zoom;

## 设计方式

`ElementNode.tsx` 做统一分发：

```tsx
export function ElementNode({ element }: Props) {
  switch (element.type) {
    case "image":
      return <ImageElementNode element={element} />;
    case "rect":
      return <RectElementNode element={element} />;
    case "text":
      return <TextElementNode element={element} />;
    case "arrow":
      return <ArrowElementNode element={element} />;
    case "group":
      return <GroupElementNode element={element} />;
    default:
      return null;
  }
}
```

`ContentLayer.tsx` 只负责：

```tsx
{elements.map((element) => (
  <ElementNode key={element.id} element={element} />
))}
```
